function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = {}; try { data = JSON.parse(e.postData.contents); } catch(err) { var params = e.parameter; for (var key in params) { data[key] = params[key]; } } sheet.appendRow([ data.fecha_registro || '', data.owner_first || data.owner_name || '', data.owner_last || '', data.email || '', data.phone || '', data.zip || '', data.pet_name || '', data.pet_species || '', data.pet_breed || '', data.pet_age || '', data.pet_weight || '', data.pet_sex || '', data.pet_neutered || '', data.pet_color || '', data.pet_cut || '', data.pet_notes || '', data.plan_seleccionado || '', data.membresia_cliente || '', data.tipo_formulario || '' ]); return ContentService .createTextOutput(JSON.stringify({result:'ok'})) .setMimeType(ContentService.MimeType.JSON); }
